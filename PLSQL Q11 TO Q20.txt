    v_principal NUMBER;
    v_annual_rate NUMBER := 9;

    v_monthly_rate NUMBER;
    v_total_months NUMBER;
    v_emi NUMBER;
    v_total_payment NUMBER;
    v_total_interest NUMBER;
    v_interest_percentage NUMBER;
